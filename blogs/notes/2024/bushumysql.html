安装mysql服务

  1. 下载压缩包，下载地址open in new window

  2. 解压下载好的压缩包

  3. 在解压目录下创建my.ini文件

    [Client]
#设置3306端口
port = 3306
[mysqld]
#设置3306端口
port = 3306
# 设置mysql的安装目录
basedir=E:/databese/mysql-5.7.44-winx64
# 设置mysql数据库的数据的存放目录
datadir=E:/databese/mysql-5.7.44-winx64/data
# 允许最大连接数
max_connections=200
# 服务端使用的字符集默认为8比特编码的latin1字符集
character-set-server=utf8
# 创建新表时将使用的默认存储引擎
default-storage-engine=INNODB
sql_mode=NO_ENGINE_SUBSTITUTION,STRICT_TRANS_TABLES
[mysql]
# 设置mysql客户端默认字符集
default-character-set=utf8

  4. 通过dos窗口进入bin目录下

    -- 创建服务，回车后会出现 Service successfully installed.
mysqld -install
-- 初始化数据目录并生成一个没有密码的root用户
mysqld --initialize-insecure --user=mysql
-- 启动mysql服务
net start mysql
-- 以root登录mysql,没有密码，直接回车
mysql -u root -p

    1. 修改密码

      # 切换到mysql数据库
use mysql;
#设置密码
update user set authentication_string=password('123456') where user='root';
# 重新加载授权表(添加用户，更改密码，修改权限都会使用到这个命令)
flush privileges;
